Quick Start

# Install client
pip install topopersist-client
# Run pipeline
curl -X POST https://api.topopersist.io/v1/pipeline/run \
-H "Authorization: Bearer YOUR_TOKEN" \
-H "Content-Type: application/json" \
-d '{"data": [[1,2,3]], "tda_module": "rips"}'

Python Example

import requests
import numpy as np

API_URL = "https://api.topopersist.io/v1"
TOKEN = "your_api_token_here"

# Generate sample data
X = np.random.rand(50, 3).tolist()
y = np.random.randint(0, 2, 50).tolist()

# Run pipeline
response = requests.post(
    f"{API_URL}/pipeline/run",
    json={
        "data": X,
        "labels": y,
        "embedding_module": "standard",
        "tda_module": "rips",
        "downstream_module": "classifier"
    },
    headers={"Authorization": f"Bearer {TOKEN}"}
)

result = response.json()
print(f"Job ID: {result['job_id']}")
print(f"Features: {result['features']}")
print(f"Accuracy: {result['downstream_scores']['mean_accuracy']}")
