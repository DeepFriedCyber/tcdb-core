Quick Start

Simple API (Unified Endpoint)

# Minimal dependencies
pip install requests numpy scipy

# Single endpoint for all descriptors
import requests

response = requests.post(
    "http://localhost:8000/descriptor/compute",
    json={
        "name": "kme_delta",
        "params": {"sigmas": [1.0, 2.0]},
        "X": [[0,0], [1,0], [0,1]]
    }
)

print(response.json())
# {'KME_delta_F': 0.42, ...}
